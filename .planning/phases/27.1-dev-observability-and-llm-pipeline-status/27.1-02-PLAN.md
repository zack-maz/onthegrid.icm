**Change 2: Wrap fetch with duration measurement.** Modify each hook's fetch function:

```typescript
// BEFORE (example: useFlightPolling)
const fetchFlights = async (): Promise<void> => {
  if (cancelled) return;
  try {
    const res = await fetch(url);
    if (cancelled) return;
    if (!res.ok) throw new Error(`Flights API ${res.status}`);
    const data = await res.json();
    setFlightData(data);
  } catch {
    if (!cancelled) setError();
  }
};

// AFTER
const fetchFlights = async (): Promise<void> => {
  if (cancelled) return;
  const start = Date.now();
  try {
    const res = await fetch(url);
    if (cancelled) return;
    if (!res.ok) {
      const msg = `Flights API ${res.status}`;
      setError(msg);
      recordFetch(false, Date.now() - start);
      return;
    }
    const data: CacheResponse<FlightEntity[]> & { rateLimited?: boolean } = await res.json();
    setFlightData(data);
    recordFetch(true, Date.now() - start);
  } catch (err) {
    if (!cancelled) {
      const msg = err instanceof Error ? err.message : 'Network error';
      setError(msg);
      recordFetch(false, Date.now() - start);
    }
  }
};
```

Key differences: (a) `const start = Date.now()` at top; (b) non-ok response does `setError(msg)` + `recordFetch(false, ...)` + `return` instead of `throw`; (c) success does `recordFetch(true, ...)` after setData; (d) catch extracts error message.
