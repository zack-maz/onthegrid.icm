      ```typescript
      new GeoJsonLayer({
        id: 'water-rivers',
        data: riversGeoJson as GeoJSON.FeatureCollection,
        getLineColor: (f: GeoJSON.Feature) => {
          // Each river feature should have properties.compositeHealth (assigned during extraction or default 0.5)
          const health = (f.properties?.compositeHealth as number) ?? 0.5;
          return stressToRGBA(health);
        },
        getLineWidth: (f: GeoJSON.Feature) => {
          const scale = (f.properties?.scalerank as number) ?? 3;
          return Math.max(1, 6 - scale) * 500;
        },
        lineWidthUnits: 'meters',
        lineWidthMinPixels: 1,
        lineWidthMaxPixels: 6,
        pickable: false, // rivers are display-only
      })
      ```
