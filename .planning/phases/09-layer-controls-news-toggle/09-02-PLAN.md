     ```typescript
     const getTooltip = useCallback((info: PickingInfo) => {
       if (!showNews || !info.object) return null;
       const entity = info.object as MapEntity;
       if (entity.type !== 'drone' && entity.type !== 'missile') return null;
       const e = entity as ConflictEventEntity;
       // Build tooltip HTML with GDELT metadata
       const lines = [
         `<strong>${e.data.eventType}</strong>`,
         e.data.locationName && `Location: ${e.data.locationName}`,
         e.data.actor1 && `Actor 1: ${e.data.actor1}`,
         e.data.actor2 && `Actor 2: ${e.data.actor2}`,
         `Date: ${new Date(e.timestamp).toISOString().slice(0, 10)}`,
         `CAMEO: ${e.data.cameoCode}`,
         `Goldstein: ${e.data.goldsteinScale.toFixed(1)}`,
         e.data.source && `<a href="${e.data.source}" target="_blank" rel="noopener" style="color:#60a5fa">Source</a>`,
       ].filter(Boolean);
       return {
         html: `<div style="font-family:monospace;font-size:11px;line-height:1.5">${lines.join('<br/>')}</div>`,
         style: {
           backgroundColor: 'rgba(0,0,0,0.85)',
           color: '#e5e5e5',
           borderRadius: '6px',
           padding: '8px 12px',
           border: '1px solid rgba(255,255,255,0.1)',
           maxWidth: '300px',
           backdropFilter: 'blur(4px)',
         },
       };
     }, [showNews]);
     ```
   - Import `PickingInfo` from `@deck.gl/core`
   - Import `MapEntity` from `@/types/entities`
   - Pass `getTooltip` to `DeckGLOverlay`: `<DeckGLOverlay layers={entityLayers} getTooltip={getTooltip} />`
   - NOTE: The tooltip HTML uses inline styles because Deck.gl's getTooltip injects a raw DOM element outside React, so Tailwind classes won't work. Use dark theme colors matching the app aesthetic.
