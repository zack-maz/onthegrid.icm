1. Create `scripts/check-env-example.ts` (~50 lines):
```typescript
#!/usr/bin/env tsx
/**
 * Drift check: every key declared in server/config.ts Zod schema
 * must appear in .env.example with a non-empty placeholder.
 *
 * Exit 1 with diff if drift found, exit 0 if clean.
 */
import { readFileSync } from 'node:fs';
import { z } from 'zod';
import { configSchema } from '../server/config.ts'; // adjust import to actual export name

function main(): number {
  const envExample = readFileSync('.env.example', 'utf8');
  const declaredKeys = new Set(
    envExample
      .split('\n')
      .map((line) => line.trim())
      .filter((line) => line && !line.startsWith('#'))
      .map((line) => line.split('=')[0]?.trim())
      .filter((k): k is string => Boolean(k)),
  );

  // Extract keys from Zod schema shape
  const schemaShape = (configSchema as z.ZodObject<z.ZodRawShape>).shape;
  const requiredKeys = Object.keys(schemaShape);

  const missing = requiredKeys.filter((k) => !declaredKeys.has(k));
  const extras = [...declaredKeys].filter((k) => !requiredKeys.includes(k));

  if (missing.length === 0 && extras.length === 0) {
    console.log('OK: .env.example matches server/config.ts schema');
    return 0;
  }

  if (missing.length > 0) {
    console.error('MISSING from .env.example:');
    missing.forEach((k) => console.error(`  - ${k}`));
  }
  if (extras.length > 0) {
    console.warn('EXTRA in .env.example (not in schema):');
    extras.forEach((k) => console.warn(`  - ${k}`));
  }
  return 1;
}

process.exit(main());
```
