<objective>
Implement the three data source adapters (OpenSky flights, AISStream ships, ACLED conflict events) with their corresponding Express routes, cache integration, and upstream API normalization to MapEntity format. Wire all routes into the Express server entry point.

Purpose: This is the core data proxying capability. Each adapter handles its upstream API's auth flow, fetches data within the Iran bounding box, normalizes to MapEntity, and caches results. Routes expose this data to the frontend via REST endpoints.
Output: Three working API endpoints (/api/flights, /api/ships, /api/events) returning normalized MapEntity arrays with staleness metadata.
</objective>

```typescript
export interface MapEntityBase {
  id: string;
  type: EntityType;
  lat: number;
  lng: number;
  timestamp: number;
  label: string;
}

export type EntityType = 'flight' | 'ship' | 'missile' | 'drone';

export interface FlightEntity extends MapEntityBase {
  type: 'flight';
  data: {
    icao24: string;
    callsign: string;
    originCountry: string;
    velocity: number | null;
    heading: number | null;
    altitude: number | null;
    onGround: boolean;
    verticalRate: number | null;
  };
}

export interface ShipEntity extends MapEntityBase {
  type: 'ship';
  data: {
    mmsi: number;
    shipName: string;
    speedOverGround: number;
    courseOverGround: number;
    trueHeading: number;
  };
}

export interface ConflictEventEntity extends MapEntityBase {
  type: 'missile' | 'drone';
  data: {
    eventType: string;
    subEventType: string;
    fatalities: number;
    actor1: string;
    actor2: string;
    notes: string;
    source: string;
  };
}

export type MapEntity = FlightEntity | ShipEntity | ConflictEventEntity;

export interface BoundingBox {
  south: number;
  north: number;
  west: number;
  east: number;
}

export interface CacheResponse<T> {
  data: T;
  stale: boolean;
  lastFresh: number;
}
```
